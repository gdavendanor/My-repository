Sub Trapecio(a,b,n,delta)
Dim N As Integer
Dim F As New clsMathParser
suma = 0
h = (b - a) / N
For i = 1 To N - 1
xi = a + i * h
suma = suma + F.Eval1(xi)
Next i
End Sub
