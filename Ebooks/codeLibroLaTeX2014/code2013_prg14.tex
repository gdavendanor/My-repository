 \begin{bteo}[Ley de Grandes Números]
    Sea $(X_n)_{n\in \mathbb{N}}$ una sucesión infinita de variables i.i.d. con
    valor esperado finito. Entonces:

    $$\frac{1}{n} \sum_{i=1}^n X_i \overset{\textnormal{a.s.}}{\longrightarrow}
    \mathbb{E} (X_1) .$$
 \end{bteo}
