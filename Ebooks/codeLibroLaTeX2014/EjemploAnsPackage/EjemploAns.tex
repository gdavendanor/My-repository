\section{Lista de ejercicios con solución }

\exercises

\exera Determine la ecuación canónica de la parábola $y=2x^2-4x+1.$
\answer  $2x^2-4x+1=y \;\Rightarrow\; 2(x-1)^2= y+1 \;\Rightarrow\;(x-1)^2=\frac{1}{2}(y+1).$
 

\exera Determine la ecuaci\'{o}n can\'{o}nica de las siguientes parábolas,\subexera $-9\,y^2 - 8\,x - 3=0$\\
  \subanswer $y^2=-\frac{8}{9}(x+\frac{3}{8})$
  %Se debe dejar un renglón
  
 \subexera $y^2+2 y-4 x=7$\\
 \subanswer $(y + 1)^2 = 4 (x + 2)$
 
  \subexera $x^2+2 x-2 y+5=0$\\
  \subanswer$(x + 1)^2 = 2 (y - 2)$

\eexer
