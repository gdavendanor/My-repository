	\begin{unadefiB}[Divisibilidad]\label{divisibilidad}
		Sean $a,b$ enteros con $b \not = 0.$ Decimos que $b$ divide a $a$
		si existe un entero $c$ tal que $a=bc.$
	\end{unadefiB}
