\begin{algorithm}[H]
\caption{Criba de Eratóstenes} \label{CribaEraostenes}
\SetAlgoLined
\KwData{$n \in\, N$}
\KwResult{Primos entre $2$ y $n$}
\LinesNumbered
max$=\left[\;\frac{(n-3)}{2}\;\right]$\;
boolean esPrimo$[i],\;\;\; i=1,2,...,max$\;
\SetAlgoVlined
\For{$j=1,2,...,max$ }{ esPrimo$[j]=$True\;}
$i=0$\;
\While{$(2i+3)(2i+3) \leq n$}{
$k=i+1$\;
\If{\rm esPrimo$[i]$}{
\While{$(2k+1)(2i+3) \leq n$}
{
esPrimo$[((2k+1)(2i+3)-3)/2]=$False\;
$k=k+1$\;
}}
$i=i+1$\; }
Imprimir\;
\For{$j=1,2,...,max$ }{
\If{\rm esPrimo$[j]=$True}{Imprima $j$ } }
\end{algorithm}
